import csv
import random

def generate_records():
    with open('data/sample_records.csv', 'r') as f:
        reader = csv.reader(f, delimiter=',')
        for row in reader:
            yield row

def predict_stroke(row):
    return row[11]

def get_random_record():
    for row in generate_records():
        yield predict_stroke(row)
